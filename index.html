val words = Strings("default")

//Puts Code on the File

words << "Hello" << "World" << "!"

val output = new mutable.HashMap[String, Int]

def count(word: String) =
if (!output.contains(word)) 0 else output(word)
//Process With Objects
words {
      word => output += word -> (count(word) + 1)
} {
      input =>
      input.keys.foreach(
           word =>
            output += word -> (input(word) + output(word)))
       output
}
